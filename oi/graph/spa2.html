queue<int>q;
bool inq[N];
int cnt[N];

	inq[s]=1;
	while(q.size()){
		int u=q.front();
		q.pop();
		inq[u]=0;
		if(cnt[u]>m)return 1;
		for(auto [v,w]:g[u]){
			if(dis[u]+w<dis[v]){
				dis[v]=dis[u]+w;
				cnt[v]=cnt[u]+1;
				if(!inq[v])q.push(v),inq[v]=1;
			}
		}
	}
	return 0;
}
